import requests
from bs4 import BeautifulSoup
import time
import csv

BASE_URL = "https://www.seniorhousingnet.com"
START_URL = "https://www.seniorhousingnet.com/seniorliving-search?state=CA"

HEADERS = {
    "User-Agent": "Mozilla/5.0 (compatible; TrueViewBot/1.0; +https://trueviewsenior.com)"
}

def get_listing_links(page_url):
    response = requests.get(page_url, headers=HEADERS)
    soup = BeautifulSoup(response.text, 'html.parser')
    links = []
    for a in soup.select("a[href*='/seniorliving-community/']"):
        href = a.get("href")
        if href and href.startswith("/seniorliving-community"):
            links.append(BASE_URL + href)
    return links

def parse_listing(url):
    response = requests.get(url, headers=HEADERS)
    soup = BeautifulSoup(response.text, 'html.parser')
    name = soup.select_one("h1")
    address = soup.select_one("div.address")
    phone = soup.select_one("div.phone-number")
    description = soup.select_one("div.description") or soup.select_one("section.about")
    images = [img["src"] for img in soup.select("img[src*='photos']") if img.get("src")]
    care_levels = [li.text.strip() for li in soup.select("ul.services li, div.services-list li")]
    amenities = [li.text.strip() for li in soup.select("ul.amenities li")]
    price = soup.select_one("div.price-range") or soup.select_one("span.price-info")

    return {
        "Name": name.text.strip() if name else "",
        "Address": address.text.strip() if address else "",
        "Phone": phone.text.strip() if phone else "",
        "Description": description.text.strip() if description else "",
        "Images": "; ".join(images),
        "Care Levels": "; ".join(care_levels),
        "Amenities": "; ".join(amenities),
        "Price": price.text.strip() if price else "",
        "Source URL": url
    }

def scrape_seniorhousingnet():
    listings_data = []
    first_page_links = get_listing_links(START_URL)

    for idx, link in enumerate(first_page_links):
        print(f"Scraping listing {idx+1}/{len(first_page_links)}: {link}")
        try:
            data = parse_listing(link)
            listings_data.append(data)
            time.sleep(2)  # Be polite
        except Exception as e:
            print(f"Error scraping {link}: {e}")
            continue

    keys = listings_data[0].keys() if listings_data else []
    with open("seniorhousingnet_sample.csv", "w", newline='', encoding="utf-8") as f:
        writer = csv.DictWriter(f, keys)
        writer.writeheader()
        writer.writerows(listings_data)

    print(f"Saved {len(listings_data)} listings to seniorhousingnet_sample.csv")

if __name__ == "__main__":
    scrape_seniorhousingnet()
