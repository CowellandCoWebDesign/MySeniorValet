// Step 1: Preprocess Community Data with Supercluster
const Supercluster = require('supercluster');

const index = new Supercluster({
  radius: 60,        // pixel radius to group points
  maxZoom: 16,       // max zoom to cluster before showing individual markers
});

// Load your data (GeoJSON format)
index.load(communitiesGeoJson.features); // communitiesGeoJson is a FeatureCollection

// Step 2: Expose a Bounding Box Endpoint for Frontend Requests
// Example Express route:
app.get('/api/communities', (req, res) => {
  const [west, south, east, north] = req.query.bbox.map(Number);
  const zoom = parseInt(req.query.zoom);

  const clusters = index.getClusters([west, south, east, north], zoom);

  res.json({
    type: 'FeatureCollection',
    features: clusters
  });
});


// ------------------------------
// FRONTEND SETUP
// ------------------------------

// Step 3: Initialize Leaflet Map
const map = L.map('map').setView([37.8, -96], 4);
const markers = L.markerClusterGroup();
map.addLayer(markers);

// Step 4: Load Clusters Based on Viewport
map.on('moveend zoomend', async () => {
  const bounds = map.getBounds();
  const zoom = map.getZoom();

  const bbox = [
    bounds.getWest(),
    bounds.getSouth(),
    bounds.getEast(),
    bounds.getNorth()
  ];

  const res = await fetch(`/api/communities?bbox=${bbox}&zoom=${zoom}`);
  const data = await res.json();

  markers.clearLayers();

  data.features.forEach(f => {
    const coords = f.geometry.coordinates;
    const props = f.properties;

    if (props.cluster) {
      const count = props.point_count;
      const clusterMarker = L.marker([coords[1], coords[0]], {
        icon: L.divIcon({
          html: `<div class="cluster">${count}</div>`,
          className: 'cluster-icon',
          iconSize: [30, 30]
        })
      });
      markers.addLayer(clusterMarker);
    } else {
      const singleMarker = L.marker([coords[1], coords[0]]);
      markers.addLayer(singleMarker);
    }
  });
});
