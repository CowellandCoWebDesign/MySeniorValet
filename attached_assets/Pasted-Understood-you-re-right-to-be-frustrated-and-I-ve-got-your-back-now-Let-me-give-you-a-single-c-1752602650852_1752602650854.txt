import React from 'react';
import { useQuery } from '@tanstack/react-query';
import Map, { Marker } from 'react-map-gl';
import 'mapbox-gl/dist/mapbox-gl.css';

const MAPBOX_TOKEN = import.meta.env.VITE_MAPBOX_TOKEN;

export default function Rentals() {
  const { data: communities = [], isLoading, error } = useQuery({
    queryKey: ['/api/communities/search', { limit: 10000 }],
    queryFn: async () => {
      const res = await fetch('/api/communities/search?limit=10000');
      if (!res.ok) throw new Error('Failed to fetch communities');
      return res.json();
    },
    staleTime: 2 * 60 * 1000,
  });

  if (isLoading) return <div className="p-4">Loading communities…</div>;
  if (error) return <div className="p-4 text-red-500">Error loading map.</div>;

  const validCommunities = communities.filter(
    (c: any) =>
      typeof c.latitude === 'number' &&
      typeof c.longitude === 'number' &&
      !isNaN(c.latitude) &&
      !isNaN(c.longitude)
  );

  return (
    <div className="w-full h-[90vh]">
      <Map
        mapboxAccessToken={MAPBOX_TOKEN}
        initialViewState={{
          latitude: 37.7749,
          longitude: -122.4194,
          zoom: 5,
        }}
        mapStyle="mapbox://styles/mapbox/streets-v12"
        style={{ width: '100%', height: '100%' }}
      >
        {validCommunities.map((community: any) => (
          <Marker
            key={community.id}
            longitude={community.longitude}
            latitude={community.latitude}
            color="blue"
          />
        ))}
      </Map>
    </div>
  );
}
