private buildLocationSearchConditions(location: string) {
  const lower = location.toLowerCase().trim();
  if (/^\d{5}$/.test(lower)) return eq(communities.zipCode, lower);
  if (lower.includes(',')) {
    const [city, state] = lower.split(',').map(s => s.trim());
    return and(eq(communities.city, city), eq(communities.state, state.toUpperCase()));
  }
  return or(
    eq(communities.city, lower),
    eq(communities.state, lower.toUpperCase())
  );
}
